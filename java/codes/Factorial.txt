package java1;

import java.util.*;

public class age{
	
	static Scanner input = new Scanner(System.in).useLocale(Locale.US);
	
	public static void main(String[] agrs)
	{
		int i, num, fact = 1;
		
		System.out.println("Enter any number : ");
		num = input.nextInt();
		
		for(i = 1; i <= num; i++)
		{
			fact *= i;
		}
		
		System.out.println("Factorial : " + fact);
			
	}
	
}
